uart

stty 9600 -F /dev/ttyS0 [-]parenb [-]parodd [-]cstopb
  • [-]parenb : parity bit
    • [-]parodd : odd parity
  • [-]cstopb : two stop bits or one with '-'
  • [-]echo : echo input characters

measure baudrate = feq * 2

echo -n -e "\125" > /dev/ttyS0

redirect

socat -d -d -x /dev/ttyS1,raw,echo=0,b115200 /dev/ttyS2,raw,echo=0,b115200


diff & patch

diff ([-u] unified mode)

diff -u file1 file2

patch ([-p1] remove the first / in path)

patch -p1 < diff.patch
