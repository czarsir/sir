    or
    git clone git://git.code.sf.net/p/openocd/code openocd
    cd openocd
    ./bootstrap nosubmodule
    ./configure --prefix=/usr/local/opt/open-ocd --enable-buspirate --enable-stlink --enable-dummy --enable-jtag_vpi --enable-remote-bitbang

GCC - avr

Install

  1. install
    2. brew tap osx-cross/avr
    brew install avr-gcc
  2. install - avr-libc
    3. http://savannah.nongnu.org/projects/avr-libc
    svn co svn://svn.savannah.nongnu.org/avr-libc/trunk
    cd trunk/avr-libc/
    ./bootstrap
    ./configure --prefix=/usr/local/Cellar/avr-gcc/8.3.0 --build=`./config.guess` --host=avr
    make
    sudo make install
    http://packs.download.atmel.com
    (gcc/dev/atmega4809/device-specs/specs-atmega4809)
    cd $PREFIX/lib/avr-gcc/8/gcc/avr/8.3.0/device-specs
    (gcc/dev/atmega4809/avrxmega3/*)
    cd $PREFIX/avr/lib/avrxmega3
    (include/avr/iom4809.h)
    cd $PREFIX/avr/include/avr
  3. install - libhidapi
    4. brew install hidapi
  4. install - avrdude
    5. http://savannah.nongnu.org/projects/avrdude
    svn co svn://svn.savannah.nongnu.org/avrdude/trunk
    cd trunk/avrdude/
    ./bootstrap
    ./configure
    make
    sudo make install
